val hand_type : hand -> hand_tp

hand_type h returns the hand type of hand h.

val hand_cards : hand -> card list

hand_cards h returns the list of cards in hand h.

val card_rank : card -> rank

card_rank c returns the rank of card c.

val card_suit : card -> suit

card_rank c returns the rank of card c.

val rank_to_int : rank -> int

rank_to_int r returns the corresponding int to a given rank Example: rank_to_int Two = 2 rank_to_int Ace = 14

val first_card : card list -> card

first_card [c1; ... ; cn] returns the first card in the list of cards c1; ... ; cn or raises an exception if the list is empty.

val last_card : card list -> card

last_card [c1; ... ; cn] returns the last card in the list of cards c1; ... ; cn or raises an exception if the list is empty.

val sub_list : 'a list -> int -> 'a list -> 'a list

sub_list [a1; ... ; an] n [a1; ... ; an] returns a sub-list of a1; ... ; an consisting of the first n items in the list, or raises an exception if the list is empty.

val same_rank : card -> card -> bool

same_rank c1 c2 returns true if cards c1 and c2 are the same rank and false if they are not.

val compare : card -> card -> int

compare c1 c2 compares cards c1 and c2, first by rank, then by suit if ranks are equal (suits ordered alphabetically), and returns a positive int if c1 is higher, a negative int if c2 is higher, or a 0 if they are equal.

val hand_compare : hand -> hand -> int

hand_compare h1 h2 compares hands h1 and h2, first by hand type, then by highest card if hands are equal, and returns a positive int if h1 is higher, a negative int if h2 is higher, or a 0 if they are equal.

val shuffle : card list -> card list

shuffle [c1; ... ; cn] randomly shuffles the list of cards c1; ... ; cn and returns the new shuffled list.

val deck : card list

deck returns the standard 52 playing card deck (without Jokers) in an unshuffled order.

val get_shuffled_deck : unit -> card list

get_shuffled_deck () returns a uniquely shuffled full deck each time it is called.

val create_player : string -> int -> int -> player

create_player name id stack creates a new active player value with name name, id id, and stack stack.

val create_players : string list -> int -> player list

create_players names stack creates a list of players for each name in names, each with an automatically generated id and stack stack
